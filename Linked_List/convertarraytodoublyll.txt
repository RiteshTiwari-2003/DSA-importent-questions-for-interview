class Node:
    def __init__(self,data):
        self.data=data
        self.prev=None
        self.next=None
def array-to_del(arr):
    if not arr:
        return None
    head=Node(arr[0])
    curr=head
    for i in range(1,len(arr)):
        new_node=Node(arr[i])
        curr.next=new_node
        new_node.prev=curr
        curr=new_node
    return head
