### insert element at the tail position in doubly linked list 

in python 

class Node:
    def __init__(self,data):
        self.data=data
        self.prev=None
        self.next=None
def insert_at_tail(head,data):
    new_node=Node(data)
    if head is None:
        return new_node
    temp=head
    while temp.next is not None:
        temp=temp.next
    temp.next=new_node
    new_node.prev=temp
    return head 
def print_list(head):
    temp=head
    while temp:
        print(temp.data,end="<->")
        temp=temp.next
    print("None")
head=Node(10)
head=insert_at_tail(head,30)
head=insert_at_tail(head,40)
print_list(head)
